\chapter{Methodolgy}

This section details the paper's approach to learn an effective policy. The method is organized into two distinct parts: Approximation of an Accurate Reward Model \ref{Approximating an Accurate Reward Model} and Iterative Policy Refinement \ref{Iterative Policy Improvement}. 

\section{Approximating and Accurate Reward Model}
\label{Approximating an Accurate Reward Model}
Before any policy learning takes place, a reward model must first be obtained. As detailed in Section \ref{RFM}, learning a reward model consists of utilizing a dataset $\mathcal{D}$ defined as $\mathcal{D}=\{(x_b,a_b,r_b)\}_{b=1}^{\|\mathcal{D}\|}$. The reward serves as the output label, and supervised learning is performed on the state-action inputs. Although this basic process can produce effective results, additional strategies are employed to further improve the accuracy and robustness of the approximate reward model. Implementing an effective data collection strategy and utilizing ensemble networks are the two primary strategies employed in this work. Together, these components help the learned reward model to produce more accurate and reliable reward estimates for state-action pairs. This accuracy is crucial, as the reward model is subsequently used to guide and optimize the policy.

\subsection{Initial Data Collection Strategy}
It is widely accepted that a model is only as good as the data on which it is trained. Ideally, a model is trained on a large and diverse dataset, enabling it to generalize across a wide range of states and actions. However, producing large-scale datasets is often time-consuming and expensive; therefore, it is essential to maximize diversity when working with limited data. To achieve this, the approach creates an initial dataset $\mathcal{D}_0$ with data generated by a human expert $\pi_{\text{Expert}}$ and a random exploration policy $\pi_{\text{Random}}$. The term ``expert" is used loosely here; while the actions should produce high rewards, they need not be strictly optimal. The purpose of the expert is to supply state-action pairs with relatively high rewards in order to provide a positive learning signal for the optimization procedure. In contrast, the random policy $\pi_{Random}$ is likely to produce lower-reward actions, but its primary purpose is to provide a broad coverage of the state and action spaces. By combining these two data sources, the constructed dataset achieves both reward diversity and state-action coverage, improving the quality of the learned reward function.

\subsection{Leveraging Ensemble Networks}
A common strategy in supervised learning is to train an ensemble of networks rather than relying on a single model. This allows multiple predictions to be produced and aggregated in ways that improve robustness and provide estimates of uncertainty. In this work, $N$ approximate reward models $\hat{r}_{\theta_1},..,\hat{r}_{\theta_N}$ are trained independently with different initial parameters $\theta_i$. The ensemble outputs are used to compute the mean predicted reward $\mu_\theta$ and the standard deviation $\sigma_\theta$, defined as 

\begin{equation}
\mu_\theta(x,a) =
\frac{1}{N} \sum_{i=1}^N
\hat{r}_{\theta_i}(x,a),
\label{Reward Mean}
\end{equation}

\begin{equation}
\sigma_\theta(x,a) =
\sqrt {\frac{1}{N} \sum_{i=1}^N
\big( \hat{r}_{\theta_i}(x,a) - \hat\mu_\theta(x,a) \big) ^2}.
\label{Reward Standard Deviation}
\end{equation}

The predicted mean reward $\mu_\theta$ serves as the primary estimate of the approximated reward. The standard deviation $\sigma_\theta$ is used quantify model uncertainty. Since independently trained models are likely to produce different predictions for state-action pairs that are poorly represented in the dataset, the standard deviation $\sigma_\theta$ provides a natural measure of uncertainty. Using these quantities, the uncertainty-penalized reward $\tilde{r}_\theta$ is defined as

\begin{equation}
\tilde{r}_\theta(x,a) =
\mu_\theta(x,a) - \beta \cdot \sigma_\theta(x,a),
\label{Penalized Reward}
\end{equation}

where the parameter $\beta$ controls the degree to which uncertainty is penalized. Larger values of $\beta$ will lead to more conservative reward approximations and will discourage the policy from selecting actions where the reward model is uncertain.

\begin{algorithm}
\caption{Train Reward Model Ensemble}
\label{alg:my_alg}
\textbf{Inputs}: Dataset $\mathcal{D}$, number of reward networks $N$
\newline
\textbf{Output}: Reward model $\tilde{r}_\theta(x,a)$
\begin{algorithmic}[1]
\State Initialize reward networks $\hat{r}_{\theta_1}, ..., \hat{r}_{\theta_N}$ with random parameters $\theta_1,..., \theta_N$
\For{$t = 1$ to $T$}
    \State {For each network $\hat r_{\theta_n}$, sample an independent 
    \Statex \hspace{0.05\linewidth} mini-batch $B_n$ from dataset $\mathcal{D}$}
    \State Update $\theta_{n} \gets \arg\min_{\theta_n}\frac{1}{B_n} \sum_{b=1}^{B_n}\mathcal{L}(r_b,\hat{r}_{\theta_n}(x_b,a_b))$
\EndFor
\State \Return $\tilde{r}_{\theta} (x,a)$
\end{algorithmic}
\end{algorithm}

\section {Iterative Policy Refinement}
\label{Iterative Policy Improvement}
Although it is common to compute a policy by directly maximizing the reward model over a set of candidate actions, this approach becomes significantly more challenging in continuous action spaces. As a result, policy optimization is adopted as the primary strategy. After training the approximate reward model, policy optimization can be carried out using the policy gradient defined in Equation \ref{GJ}, which has been shown to be an effective approach for learning policies. However, this work employs a modified learning objective together with an iterative policy learning process to achieve improved policy performance and greater robustness compared to standard optimization.

\subsection{Maximum Action Noise Threshold Objective}
The main contribution of this paper is a modified learning objective. In reinforcement learning, the main objective is to maximize expected reward, formalized in Equation \ref{J}. However, when optimizing  this objective in the offline setting with an approximate reward model, two issues can arise. First, the policy may exploit inaccuracies in the reward model, selecting actions that are highly rated by the approximate reward function but are actually suboptimal in the real-world. Secondly, the policy may converge to local optima due to limited exploration and the absence of online feedback. To address these challenges, we introduce a new learning objective based on a function $\eta(x,a)$, called the \textbf{Maximum Action Noise Threshold (MaxANT)} function, defined as 

\begin{equation}
\label{Max ANT Function}
\eta(x,a) = 
\eta^+(x,a) + \eta^-(x,a),
\end{equation}

Intuitively, $\eta(x,a)$ measures the maximum positive and negative perturbation $\Delta$ that can be applied to action $a$ without the reward falling below the reward threshold $\tau \in [r_{\text{min}}, r_{\text{max}}]$. This function is incorporated into the standard objective and is multiplied by the hyperparameter $\alpha$ to produce the modified objective

\begin{equation}
\label{New Objective}
J(\phi) =
\mathbb{E}_{x \sim p_x(\cdot)}
% [r(x,\pi_\phi(x)) + \alpha \mathcal{F}(x, \pi_\phi(x),\tau)],
\big[r\big(x,\pi_\phi(x)\big) + \alpha \cdot\eta\big(x, \pi_\phi(x)\big)\big],
\end{equation}

where the policy $\pi_\phi$ now aims to maximize not only expected reward, but also robustness to action perturbations. This encourages the selection of actions that not only have high reward but also lie in regions where nearby actions are similarly rated. As a result, the learned policy $\pi_\phi$ is less likely to exploit narrow peaks in the reward landscape and is more likely to converge to broad, high reward regions. This mitigates the risk of both reward exploitation and convergence to local optima. The standard RL objective is recovered as $\alpha$ approaches zero. The policy gradient $\nabla_\phi J(\phi)$ for this objective is

\begin{equation}
\label{New Policy Gradient}
\nabla_\phi J(\phi) =
\mathbb{E}_{x \sim p_x(\cdot)}
\big[\nabla_\phi \pi_\phi(x) \cdot 
\nabla_a \big(r(x,a) + \alpha \cdot \eta(x,a)\big) |_{a=\pi_\phi(x)} \big],
\end{equation}

which can be approximated by 

\begin{equation}
\label{Aproximate New Policy Gradient}
\nabla_\phi J(\phi) =
\frac{1}{B} \sum_{b=1}^{B}\nabla_\phi \pi_\phi(x_b) 
\cdot \nabla_a\big(r(x_b,a_b) + \alpha \cdot \eta(x_i,a_i)\big) |_{a_i=\pi_\phi(x_i)}.
\end{equation}

where $B$ represents the batch size of the sampled states. 

\subsection{Iterative Deployment of Trained Policy}
After training the policy with the modified objective described above, the learned policy $\pi_\phi$ is deployed in the real world to generate a dataset $\mathcal{D}_{\pi_\phi}$. Since the learned policy $\mathcal{D}_{\pi_\phi}$ is deterministic, Gaussian noise is added to the continuous actions during its deployment to evaluate its robustness. Once collected, $\mathcal{D}_{\pi_\phi}$ is merged with the existing dataset to form $\mathcal{D}_{i+1} = \mathcal{D}_i \ \cup \ \mathcal{D}_{\pi_\phi}$. The resulting dataset now contains a mixture of random exploration, expert, and on-policy data, increasing overall diversity. The new dataset is subsequently used to retrain the reward model, which in turn is used to further optimize the policy. By augmenting the dataset with on-policy samples, the reward model is refined in regions of the state-action space that are most relevant to the current policy. This helps correct suboptimal policy convergence that may arise from optimizing with an imperfect reward model though evaluation with real world interactions. The full algorithm is presented in Figure \ref{Overall Algorithm} and Algorithm \ref{alg:my_alg}.

\begin{algorithm}
\caption{Iterative Policy Improvement with MaxANT RL}
\label{alg:my_alg}
\textbf{Inputs}: Number of reward networks $N$, action noise threshold $\tau$, action noise threshold rate $\alpha$
\newline
\textbf{Output}: Policy $\pi_\phi$
\begin{algorithmic}[1]
\State Initialize policy network $\pi_\phi$ with random parameters $\phi$
\State Fill initial dataset $\mathcal{D}_0$ with data from random exploration policy $\pi_\text{Random}$ and from expert policy $\pi_\text{Expert}$
\For{$i = 0$ to $I$}
    \State $\tilde{r}_{\theta}(x,a) \gets$ TrainRewardModel$(\mathcal{D}_i,N)$
    \For{$t = 1$ to $T$}
        \State Sample mini-batch of size $B$ from dataset $\mathcal{D}_i$
        \State \makebox[\linewidth][l]{$\nabla_\phi J(\phi) =  \frac{1}{B} \sum_{b=1}^{B}\nabla_\phi \pi_\phi(x_b) \cdot \nabla_a\big(\tilde{r}(x_b,a_b) + \alpha \cdot \eta(x_b,a_b)\big) |_{a_i=\pi_\phi(x_b)}$}
        \State Update $\phi$ using Equation \ref{Aproximate New Policy Gradient}
    \EndFor
    \State Execute $\pi_{\phi_i}$ and add interactions to $\mathcal{D}_{\pi_{\phi_i}}$
    \State Update dataset $\mathcal{D}_{i+1}=\mathcal{D}_i \cup \mathcal{D}_{\pi_{\phi_i}}$
\EndFor
\State \Return $\pi_\phi$
\end{algorithmic}
\end{algorithm}
